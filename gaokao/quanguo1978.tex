\begin{questions}
	\question
	\begin{parts}
		\part 分解因式:$x^2 - 4xy + 4y^2 - 4z^2$.

			\begin{solution}
				\begin{align*}
					 & = (x-2y)^2 - 4z^2    \\
					 & = (x-2y+2z)(x-2y-2z)
				\end{align*}
			\end{solution}

	\end{parts}

\end{questions}
