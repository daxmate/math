\section[1978年高考数学试卷(备用卷)文理]{1978年普通高等学校招生考试(备用卷)\\\Huge{数学试卷}}
\begin{questions}
	\question
	\begin{parts}
		\part 分解因式:$x^2 - 2xy + y^2 + 2x - 2y - 3$.

			\begin{solution}
				\begin{align*}
					 & = (x-y)^2 + 2(x-y) - 3 \\
					 & = (x-y +3)(x-y-1)
				\end{align*}
			\end{solution}

	\end{parts}
\end{questions}
